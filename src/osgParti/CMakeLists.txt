set(LIB_NAME osgParti)
set(HEADER_PATH ${CMAKE_SOURCE_DIR}/include/${LIB_NAME})
FILE(GLOB HEADERS "${HEADER_PATH}/*.h*")
FILE(GLOB SOURCES "*.cpp")

set(SOURCES ${SOURCES})

include_directories(
	${CMAKE_SOURCE_DIR}/include
	${OSG_INCLUDE_DIR}
	${QT_INCLUDE_DIR} 
)

IF   (QT4_FOUND)
    SET(TARGET_EXTERNAL_LIBRARIES ${QT_QTCORE_LIBRARY_RELEASE} ${QT_QTGUI_LIBRARY_RELEASE} ${QT_QTOPENGL_LIBRARY_RELEASE} )
    ADD_DEFINITIONS(-DUSE_QT4)
ELSE(QT4_FOUND)
    SET(TARGET_EXTERNAL_LIBRARIES ${QT_LIBRARIES} )
ENDIF(QT4_FOUND)



SET(LIBRARIES_OPTIMIZED 
	${OSG_LIBRARIES} 
	${OPENGL_LIBRARY} 
	${CMAKE_THREAD_LIBS_INIT} 
)
SET(LIBRARIES_DEBUG 
	${OSG_LIBRARIES_DEBUG} 
	${OPENGL_LIBRARY} 
	${CMAKE_THREAD_LIBS_INIT} 
)

SETUP_CORELIB(${LIB_NAME})
